• git [команда]локальный
  • git --version [подкоманда]версия git
  • git config --global user.name "[имя]"
  • git config --global user.email "[адрес электронной почты]"
  • git config --list --show-origin [проверка подключения]
  • git init [инициализация папки для работы с git]
  • git add . [индексировать изменения файла,все изменения] git add [файл] [индексировать определеные изменения]
  • git commit -m "[сообщение с описанием]"[фиксирует и сохроняет их в историю]
  • git status[дает возтожность просмотреть изменения на данном этапе нашего проэкта.]
  • git log [показывает все комиты]
  • git reset --soft [id путь] [возврат к существующим но не проиндексированым файлам.] git reset --hard [удаление файла до следующего комита.]
  • git push [Загружает все изменения локальной ветки в удалённый репозиторий]
  • git pull [Загружает историю из удалённого репозитория и объединяет её с локальной]
  • git branch [Список именованных веток коммитов с указанием выбранной ветки]
